import matplotlib.mlab as mlab
from random import shuffle

sigma_coefs = [640, 225, 80]

for sigma in sigma_coefs:
    x = 2500+sigma*np.random.randn(10000)
    x = x.astype(int)
    x = list(x)
    x.extend(X)
    shuffle(x)
    distributions.append(x)
